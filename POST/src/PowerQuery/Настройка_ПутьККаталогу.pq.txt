// Запрос «Настройка_ПутьККаталогу»
// Назначение: вернуть текстовый путь к папке, где лежит книга Excel и файл «Отбор.csv».
//
// В книге Excel создайте именованную область «КаталогРабочейКниги» (одна ячейка) с формулой:
//   =ЛЕВСИМВ(ЯЧЕЙКА("имяфайла";A1);ПОИСК("[";ЯЧЕЙКА("имяфайла";A1))-1)
// (для русской локали разделитель в формулах — «;».)
//
// Результат должен заканчиваться обратным слэшем «\», например: D:\Данные\Проект\

let
    // Читаем одноимённую таблицу или именованную область из текущей книги
    Источник = Excel.CurrentWorkbook(){[Name = "КаталогРабочейКниги"]}[Content],
    // Берём значение из первой строки первого столбца (имя столбца может отличаться от «Column1»)
    ИмяПервогоСтолбца = Table.ColumnNames(Источник){0},
    ПутьСырой = Record.Field(Источник{0}, ИмяПервогоСтолбца),
    ПутьТекст = Text.Trim(Text.From(ПутьСырой)),
    // Нормализуем завершающий слэш для Windows
    СоСлэшем =
        if Text.EndsWith(ПутьТекст, "\") then
            ПутьТекст
        else
            ПутьТекст & "\"
in
    СоСлэшем
